/* ── ALIAS SNAPSHOT LOADER (MSG158) ────────────────────────────
   Loads alias snapshot from service or falls back to embedded.
   Must be called before resolvers are used. */
function buildAliasMapsFromRows(rows) {
  const regionMap = {};
  const ptAliases = [];
  for (const row of rows) {
    const type = String(row.entity_type || '').toLowerCase();
    const canon = String(row.canonical_id || '').trim();
    const alias = String(row.alias || '').trim();
    if (!type || !canon || !alias) continue;
    if (type === 'region') {
      const canonLow = canon.toLowerCase();
      regionMap[alias.toLowerCase()] = canonLow;
      regionMap[canonLow] = canonLow;  /* self-mapping for canonical id */
    } else if (type === 'product_type') {
      ptAliases.push(alias.toLowerCase());
    }
  }
  return { regionMap, ptAliases };
}

/* MSG173: CDN JSON fetch for alias snapshot.
   GitHub Pages serves static JSON — no CORS issues, no postMessage needed.
   Simple fetch with timeout + fallback to embedded snapshot. */
async function loadAliasSnapshot() {
  _aliasState.transport = 'none';

  if (ALIAS_SNAPSHOT_CONFIG.enabled && ALIAS_SNAPSHOT_CONFIG.url) {
    try {
      const ctrl = new AbortController();
      const timer = setTimeout(() => ctrl.abort(), ALIAS_SNAPSHOT_CONFIG.timeout);
      const resp = await fetch(ALIAS_SNAPSHOT_CONFIG.url, {
        signal: ctrl.signal,
        cache: 'no-cache',  /* always get latest version */
      });
      clearTimeout(timer);
      if (resp.ok) {
        const data = await resp.json();
        if (data && data.ok && Array.isArray(data.rows)) {
          const maps = buildAliasMapsFromRows(data.rows);
          _aliasState.source = 'live_cdn_snapshot';
          _aliasState.transport = 'cdn-json';
          _aliasState.version = data.version || '';
          _aliasState.updatedAt = data.updated_at || '';
          _aliasState.rowCount = data.rows.length;
          _aliasState.regionMap = maps.regionMap;
          _aliasState.ptAliases = maps.ptAliases;
          _aliasState.loaded = true;
          _aliasState.error = '';
          _aliasState.status = 'LIVE_OK';
          return;
        }
        _aliasState.error = 'CDN_RESPONSE_INVALID: ' + JSON.stringify(data).slice(0, 200);
        _aliasState.status = 'CDN_RESPONSE_INVALID';
      } else {
        _aliasState.error = 'CDN_HTTP_' + resp.status;
        _aliasState.status = 'CDN_HTTP_ERROR';
      }
    } catch (e) {
      const msg = e.message || String(e);
      if (msg.includes('abort')) {
        _aliasState.error = 'CDN_TIMEOUT: ' + ALIAS_SNAPSHOT_CONFIG.timeout + 'ms';
        _aliasState.status = 'CDN_TIMEOUT';
      } else {
        _aliasState.error = 'CDN_FETCH_ERROR: ' + msg;
        _aliasState.status = 'CDN_FETCH_ERROR';
      }
    }
  } else {
    _aliasState.error = 'CDN URL not configured';
    _aliasState.status = 'CDN_NOT_CONFIGURED';
  }

  /* Fallback to embedded snapshot */
  const maps = buildAliasMapsFromRows(EMBEDDED_ALIAS_SNAPSHOT_FALLBACK.rows);
  _aliasState.source = 'embedded_fallback';
  _aliasState.transport = 'fallback';
  _aliasState.version = EMBEDDED_ALIAS_SNAPSHOT_FALLBACK.version;
  _aliasState.updatedAt = '';
  _aliasState.rowCount = EMBEDDED_ALIAS_SNAPSHOT_FALLBACK.rows.length;
  _aliasState.regionMap = maps.regionMap;
  _aliasState.ptAliases = maps.ptAliases;
  _aliasState.loaded = true;
  _aliasState.status = 'FALLBACK_USED';
}

function canonicalRegion(r) {
  const k = String(r || '').toLowerCase().trim();
  /* MSG158: prefer snapshot-driven map, fall back to compile-time map */
  if (_aliasState.loaded && _aliasState.regionMap[k]) return _aliasState.regionMap[k];
  return REGION_ALIASES[k] || k;
}
function regionsCanonicallyEqual(a, b) {
  if (!a || !b) return false;
  /* Compare comma-separated region lists */
  const listA = String(a).split(',').map(s => canonicalRegion(s.trim())).filter(Boolean).sort();
  const listB = String(b).split(',').map(s => canonicalRegion(s.trim())).filter(Boolean).sort();
  return listA.join(',') === listB.join(',');
}

/* ── CANONICAL PRODUCT TYPE (MSG84 + MSG137 catalogue-driven) ── */
/* Canonicalize by: lowercase, split on separators, sort parts, join.
   This is the basic heuristic layer — catalogue resolver takes priority. */
function canonicalProductType(v) {
  const s = String(v || '').toLowerCase().trim();
  if (!s) return '';
  /* Split on " - ", " – " (display separators), NOT on bare "-" (part of slugs/brand names) */
  const parts = s.split(/\s+[-–]\s+/).map(p => p.trim()).filter(Boolean).sort();
  return parts.join('::');
}

/* MSG137: Catalogue-driven ProductType resolver.
   Uses popup productTypeOptions as BSV catalogue source of truth.
   The catalogue slug from backend (e.g. "games-ea-sports-fc") maps to a
   display label in the popup (e.g. "EA Sports FC - Games").
   Both refer to the same catalogue entity. */

/* Normalize slug for comparison: lowercase, replace [-_] with spaces, collapse whitespace */
function normSlug(s) {
  return String(s || '').toLowerCase().replace(/[-_]+/g, ' ').replace(/\s+/g, ' ').trim();
}

/* Find popup option that matches a candidate slug/id.
   Returns {value, label} or null.
   Match strategy: slug-normalized comparison of option labels. */
function resolveProductTypeFromCatalogue(candidateId, options) {
  if (!candidateId || !options || !options.length) return null;
  const needle = normSlug(candidateId);
  if (!needle) return null;
  /* 1. Exact label match (norm) */
  for (const opt of options) {
    if (norm(opt.label) === norm(candidateId)) return opt;
  }
  /* 2. Slug-normalized match: "games-ea-sports-fc" ↔ "EA Sports FC - Games" */
  for (const opt of options) {
    if (normSlug(opt.label) === needle) return opt;
  }
  /* 3. Sorted-parts match: split both into words, sort, compare */
  const needleParts = needle.split(/\s+/).sort().join(' ');
  for (const opt of options) {
    const optParts = normSlug(opt.label).split(/\s+/).sort().join(' ');
    if (optParts === needleParts) return opt;
  }
  return null;
}

/* Check if candidate is a generic platform family (e.g., Gift_cards-Xbox) */
function isGenericPlatformCandidate(candidateSlug) {
  const s = normSlug(candidateSlug);
  return PT_GENERIC_FAMILIES.some(fam => s.includes(fam)) &&
         PT_PLATFORM_KEYWORDS.some(plat => s.includes(plat));
}

/* Check if text contains strong product identity signals */
function findIdentitySignals(text) {
  const t = String(text || '').toLowerCase();
  return PT_IDENTITY_KEYWORDS.filter(kw => t.includes(kw));
}

/* Check if text's only relevant signal is a platform keyword */
function findPlatformSignals(text) {
  const t = String(text || '').toLowerCase();
  return PT_PLATFORM_KEYWORDS.filter(kw => {
    /* Match as whole word to avoid false positives (e.g., "switch" in "switching") */
    const re = new RegExp('\\b' + kw.replace(/[.*+?^${}()|[\]\\]/g, '\\$&') + '\\b');
    return re.test(t);
  });
}

function resolveProductTypeState(currentLabel, candidateId, ptOptions, moduleStatus, productName) {
  const state = {
    currentRaw: currentLabel || '',
    recommendedRaw: candidateId || '',
    currentDisplay: currentLabel || '',
    recommendedDisplay: candidateId || '',  /* default: show slug */
    currentCanonical: canonicalProductType(currentLabel),
    recommendedCanonical: canonicalProductType(candidateId),
    semanticMatch: false,
    catalogueResolved: false,
    candidateSuspect: false,       /* MSG142: true if candidate looks like false platform match */
    identitySignals: [],           /* MSG142: identity keywords found in product name */
    platformSignals: [],           /* MSG142: platform keywords found in product name */
    suspectReason: '',             /* MSG142: why candidate is suspect */
    showHelper: false,
    helperMode: 'none',       /* 'none' | 'replacement' | 'unresolved' */
    uiStatus: 'yellow',       /* 'green' | 'yellow' | 'red' */
  };

  if (!candidateId) {
    state.semanticMatch = true;
    state.uiStatus = 'green';
    return state;
  }

  /* Step 1: Try catalogue-driven resolution */
  const catalogueMatch = resolveProductTypeFromCatalogue(candidateId, ptOptions || []);
  if (catalogueMatch) {
    state.catalogueResolved = true;
    state.recommendedDisplay = catalogueMatch.label;
    state.recommendedCanonical = canonicalProductType(catalogueMatch.label);
  }

  /* Step 2: Check semantic match */
  /* Quick exact/norm check */
  if (currentLabel === candidateId || norm(currentLabel) === norm(candidateId)) {
    state.semanticMatch = true;
  }
  /* Canonical check (display labels) */
  else if (state.currentCanonical && state.recommendedCanonical &&
           state.currentCanonical === state.recommendedCanonical) {
    state.semanticMatch = true;
  }
  /* Catalogue-resolved display label vs current */
  else if (catalogueMatch && norm(currentLabel) === norm(catalogueMatch.label)) {
    state.semanticMatch = true;
  }
  /* Catalogue-resolved: slug-normalized current vs resolved label */
  else if (catalogueMatch && normSlug(currentLabel) === normSlug(catalogueMatch.label)) {
    state.semanticMatch = true;
  }
  /* Backend says match */
  else if (moduleStatus === 'match') {
    state.semanticMatch = true;
  }

  /* Step 2b (MSG142): Candidate quality validation — identity vs platform signal priority.
     If candidate is a generic platform family (e.g., Gift_cards-Xbox) but product name
     contains strong identity signals (e.g., "EA Sports FC", "Edition"), the candidate
     is likely a false positive driven by a platform keyword. */
  if (!state.semanticMatch && productName) {
    state.identitySignals = findIdentitySignals(productName);
    state.platformSignals = findPlatformSignals(productName);
    const candidateIsGenericPlatform = isGenericPlatformCandidate(candidateId);

    if (candidateIsGenericPlatform && state.identitySignals.length > 0) {
      /* Product has strong identity signals but candidate is generic platform → suspect */
      state.candidateSuspect = true;
      state.suspectReason = 'candidate is generic platform family (' +
        (state.catalogueResolved ? state.recommendedDisplay : candidateId) +
        ') but product name contains identity signals: ' +
        state.identitySignals.join(', ') +
        '. Platform keyword alone insufficient.';
    }

    /* Also check: if current PT already contains identity keywords that match product name,
       and candidate is a completely different family, this strengthens suspicion */
    if (!state.candidateSuspect && candidateIsGenericPlatform) {
      const currentIdentity = findIdentitySignals(currentLabel);
      const nameIdentity = findIdentitySignals(productName);
      const overlap = currentIdentity.filter(kw => nameIdentity.includes(kw));
      if (overlap.length > 0) {
        state.candidateSuspect = true;
        state.suspectReason = 'current PT shares identity signals with product name (' +
          overlap.join(', ') + ') but candidate is generic platform family.';
      }
    }
  }

  /* Step 3: Determine UI state */
  if (state.semanticMatch) {
    state.uiStatus = 'green';
    state.showHelper = false;
    state.helperMode = 'none';
  } else if (state.candidateSuspect) {
    /* MSG142: suspect candidate → yellow but DO NOT show replacement helper.
       Instead show unresolved/warning — operator should verify manually. */
    state.uiStatus = 'yellow';
    state.showHelper = true;
    state.helperMode = 'unresolved';
  } else if (moduleStatus === 'mismatch') {
    state.uiStatus = 'red';
    state.showHelper = true;
    state.helperMode = 'replacement';
  } else {
    state.uiStatus = 'yellow';
    state.showHelper = true;
    state.helperMode = 'replacement';
  }

  return state;
}

function productTypesCanonicallyEqual(current, recommended) {
  if (!current || !recommended) return false;
  if (current === recommended || norm(current) === norm(recommended)) return true;
  return canonicalProductType(current) === canonicalProductType(recommended);
}

/* ── CANONICAL DENOMINATION (bugfix MSG79) ───────────────────── */
/* DOM select values use format: [TypeKey][mode][Category][value]
   Backend candidate IDs use format: dn::TypeKey::value
   Both refer to the same denomination entity.
   Extract entity key for canonical comparison. */
function extractDenomEntity(v) {
  const s = String(v || '').trim();
  if (!s) return '';
  /* Format: dn::SomeKey::Value → key=SomeKey, val=Value */
  const dnMatch = s.match(/^dn::(.+?)::(.+)$/i);
  if (dnMatch) return (dnMatch[1] + '::' + dnMatch[2]).toLowerCase();
  /* Format: [TypeKey][mode][Category][value] → key=TypeKey, val=value */
  const brMatch = s.match(/^\[([^\]]+)\]\[([^\]]*)\]\[([^\]]*)\]\[([^\]]*)\]$/);
  if (brMatch) return (brMatch[1] + '::' + brMatch[4]).toLowerCase();
  /* Fallback: normalize */
  return s.toLowerCase().replace(/[_\-\s]+/g, '_');
}
function denominationsCanonicallyEqual(current, recommended) {
  if (!current || !recommended) return false;
  /* Quick exact/norm check first */
  if (current === recommended || norm(current) === norm(recommended)) return true;
  /* Entity extraction */
  const eCur = extractDenomEntity(current);
  const eRec = extractDenomEntity(recommended);
  return eCur && eRec && eCur === eRec;
}
/* Find denomination <option> by entity match (bugfix MSG119).
   candidateId is in backend format (dn::TypeKey::value).
   Options are in DOM format ([TypeKey][mode][Category][value]).
   Uses extractDenomEntity for canonical matching. */
function findDenominationOptionByEntity(selectEl, candidateId) {
  if (!selectEl || !candidateId) return null;
  const needleEntity = extractDenomEntity(candidateId);
  if (!needleEntity) return null;
  for (const opt of selectEl.options) {
    if (!opt.value) continue;
    const optEntity = extractDenomEntity(opt.textContent.trim());
    if (optEntity && optEntity === needleEntity) return opt;
  }
  return null;
}

/* ── SAFETY GATES (Stage 1.1 + 1.2 patch) ───────────────────── */
/* Blocking applyError patterns — these indicate unresolved manual steps.
   Card with any of these MUST NOT be batch-saved without operator resolution.
   Stage 1.2: manuallyConfirmed=true clears blocking issues after operator
   explicitly confirms manual fixes via "Подтвердить ручные правки" action. */
const BLOCKING_APPLY_PATTERNS = ['REGION_MANUAL', 'PT_NO_MATCH', 'DENOM_NO_MATCH'];

function getCardBlockingIssues(card) {
  const issues = [];
  if (!card.applied) { issues.push('not_applied'); return issues; }
  if (card.saved)    return [];   /* already saved — no blockers */
  if (card.manuallyConfirmed) return [];   /* Stage 1.2: operator confirmed manual fixes */
  if (card.applyError) {
    for (const pat of BLOCKING_APPLY_PATTERNS) {
      if (card.applyError.includes(pat)) issues.push(pat);
    }
  }
  return issues;
}

function isCardSaveReady(card) {
  return card.applied && !card.saved && getCardBlockingIssues(card).length === 0;
}

function getModStatusBlockingReason(card) {
  /* green: mod status → published, allowed without save */
  if (card.finalColor === 'green') return null;
  /* red: mod status → rejected, allowed (reason 41 configured) */
  if (card.finalColor === 'red') return null;
  /* yellow: mod status = no_change, so always "allowed" (it's a no-op) */
  if (card.finalColor === 'yellow') return null;
  /* error: no mod status action */
  if (card.finalColor === 'error') return 'error_card';
  return null;
}

function isCardModStatusReady(card) {
  const rule = getModStatusRule(card.finalColor);
  if (!rule || rule._blocked || rule.modStatus === 'no_change') return false;
  return getModStatusBlockingReason(card) === null;
}

function openWorkerWindow(url) {
  if (_s.workerWin && !_s.workerWin.closed) {
    try { _s.workerWin.close(); } catch (_) {}
  }
  const w = window.open(url, '__bsv_wip01_worker__',
    'width=1100,height=760,left=80,top=80,scrollbars=yes,resizable=yes');
  if (!w) throw new Error(
    'POPUP_BLOCKED: Браузер заблокировал popup-окно.\n' +

  );

}

async function navigateWorkerWindow(win, url) {
  const seq = ++_s.navSeq;
  try { win.__bsv_nav_seq__ = seq; } catch (_) {}
  win.location.href = url;
  const deadline = Date.now() + NAV_TIMEOUT_MS;
  while (Date.now() < deadline) {
    if (win.closed) throw new Error('WORKER_CLOSED: Popup закрыт во время навигации');
    try { if (win.__bsv_nav_seq__ !== seq) break; } catch (_) { break; }
    await wait(100);
  }
  await wait(200);
}

async function waitForWorkerReady(win, ms) {
  const deadline = Date.now() + (ms || PAGE_LOAD_MS);
  while (Date.now() < deadline) {
    if (win.closed) throw new Error('WORKER_CLOSED: Popup закрыт во время ожидания страницы');
    try {
      const doc = win.document;
      if (doc && doc.readyState !== 'loading' && doc.querySelector('#product-form')) return;
    } catch (_) {}
    await wait(POLL_MS);
  }
  throw new Error('WORKER_PAGE_TIMEOUT: Страница не загрузилась за ' +
    Math.round((ms || PAGE_LOAD_MS) / 1000) + 'с');
}

async function ensureWorkerAtUrl(url) {
  const fullUrl = /^https?:\/\//i.test(url) ? url : (BASE + url);
  if (_s.isFirstCard || !_s.workerWin || _s.workerWin.closed) {
    openWorkerWindow(fullUrl);
    _s.isFirstCard = false;
  } else {
    await navigateWorkerWindow(_s.workerWin, fullUrl);
  }
  await waitForWorkerReady(_s.workerWin, PAGE_LOAD_MS);
  return _s.workerWin;
}

/* ── BACKEND CALLER (postmessage iframe transport) ───────────── */
function callBackend(payload) {
  return new Promise(resolve => {
    const reqId  = 'wip01_' + Date.now() + '_' + Math.random().toString(36).slice(2);
    const iframe = document.createElement('iframe');
    const form   = document.createElement('form');
    iframe.name       = 'wip01_f_' + reqId;
    iframe.style.display = 'none';
    form.method       = 'POST';
    form.action       = BACKEND_URL;
    form.target       = iframe.name;
    form.style.display = 'none';

    const addInput = (n, v) => {
      const i = document.createElement('input');
      i.type = 'hidden'; i.name = n;
      i.value = (v == null) ? '' : String(v);
      form.appendChild(i);
    };
    addInput('transport',         'postmessage');
    addInput('client_request_id', reqId);
    addInput('payload_json',      JSON.stringify(payload || {}));

    let settled = false;
    const tId = setTimeout(() => {
      if (settled) return;
      settled = true; cleanup();
      resolve({ ok: false, final_status: 'review', reason: 'AI Core timeout (50s)', _timeout: true });
    }, BACKEND_TIMEOUT);

    function cleanup() {
      const fn = _s.msgHandlers.get(reqId);
      if (fn) { window.removeEventListener('message', fn); _s.msgHandlers.delete(reqId); }
      clearTimeout(tId);
      setTimeout(() => {
        try { form.remove();   } catch (_) {}
        try { iframe.remove(); } catch (_) {}
      }, 600);
    }

    const handler = ev => {
      const d = ev && ev.data;
      if (!d || typeof d !== 'object') return;
      if (d.source !== POSTMESSAGE_SRC) return;
      if (String(d.client_request_id || '') !== reqId) return;
      if (settled) return;
      settled = true; cleanup(); resolve(d);
    };
    _s.msgHandlers.set(reqId, handler);
    window.addEventListener('message', handler);
    document.body.appendChild(iframe);
    document.body.appendChild(form);
    try { form.submit(); } catch (e) {
      cleanup();
      resolve({ ok: false, final_status: 'review', reason: 'Form submit error: ' + e.message });
    }
  });
}

/* ── BACKEND RESULT MAPPER ───────────────────────────────────── */
/* Maps Apps Script response to internal card verdict + recommendations.
   INTEGRATION NOTE: backend returns catalogue candidate_id keys (text),
   not DOM select option values. resolvedDomValue is filled during apply
   via best-effort text matching. If no match → applyError set, field not changed. */
function mapBackendResult(raw) {
  if (!raw || raw._timeout) {
    return {
      verdict: 'error',
      confidence: null,
      reason: raw ? raw.reason : 'Нет ответа от сервера',
      recommendations: null,
    };
  }
  const fs = String(raw.final_status || '').toLowerCase();
  let verdict = 'error';
  if (fs === 'approve') verdict = 'green';
  else if (fs === 'review') verdict = 'yellow';
  else if (fs === 'reject') verdict = 'red';

  const sel     = raw.selection || {};
  const modules = raw.modules   || {};

  return {
    verdict,
    confidence:      raw.confidence != null ? Number(raw.confidence) : null,
    reason:          norm(raw.reason || ''),
    recommendations: {
      productType: {
        status:           (modules.product_type || {}).status || 'unknown',
        candidateId:      sel.product_type_candidate_id || '',
        resolvedDomValue: null,   /* filled during applyRecommendations() */
        resolvedDomLabel: null,
      },
      region: {
        status:      (modules.region || {}).status || 'unknown',
        candidateId: sel.region_candidate_id || '',
        /* Stage 1: region auto-apply not supported (needs candidate→region_id mapping).
           Flagged as REGION_MANUAL in applyError if status !== 'match'. */
      },
      denomination: {
        status:           (modules.denomination || {}).status || 'unknown',
        candidateId:      sel.denomination_candidate_id || '',
        resolvedDomValue: null,
        resolvedDomLabel: null,
      },
    },
  };
}

/* ── SUPPRESS PRODUCT TYPE AUTO-SUBMIT (W3) ──────────────────── */
/* Clone-replace the select to remove all jQuery event listeners.

  }

/* ── BEST-EFFORT TEXT MATCH (for apply) ──────────────────────── */
function findOptionByText(selectEl, candidateId) {
  if (!selectEl || !candidateId) return null;
  const needle = candidateId.toLowerCase().replace(/_/g, ' ').replace(/-/g, ' ').trim();
  let best = null;
  for (const opt of selectEl.options) {
    if (!opt.value) continue;
    const txt = opt.textContent.toLowerCase().trim();
    if (txt === needle) return opt;                          /* exact match */
    const parts = txt.split(/\s[-–]\s/);                    /* "Brand - Category" */
    if (parts[0] && parts[0].trim() === needle) { best = opt; continue; }
    if (txt.includes(needle) || needle.includes(parts[0] || '')) {
      if (!best) best = opt;
    }
  }
  return best;
}

      denomination: snap.denominationLabel || '',

}

/* ── APPLY SINGLE FIELD in popup DOM ────────────────────────── */
function applyProductTypeInPopup(win, candidateId) {
  if (!candidateId) return { ok: false, error: 'PT_EMPTY: Нет candidateId' };
  const ptSel = suppressProductTypeAutoSubmit(win);
  if (!ptSel) return { ok: false, error: 'PT_NO_SELECT: select не найден' };
  const matched = findOptionByText(ptSel, candidateId);
  if (!matched) return { ok: false, error: 'PT_NO_MATCH: ProductType кандидат «' + candidateId + '» не найден в select DOM' };
  ptSel.value = matched.value;
  return { ok: true, value: matched.value, label: norm(matched.textContent) };
}

function applyDenominationInPopup(win, candidateId) {
  /* MSG121: full diagnostic instrumentation for denomination apply */
  const diag = {
    ts: isoNow(),
    candidateRaw: candidateId || '',
    candidateEntity: '',
    selectFound: false,
    optionCount: 0,
    options: [],          /* [{value, label, entity}] — all popup options */
    currentRaw: '',
    currentEntity: '',
    currentValue: '',
    alreadyCanonEqual: false,
    matchedOptionValue: '',
    matchedOptionLabel: '',
    matchedOptionEntity: '',
    oldSelectValue: '',
    newSelectValue: '',
    valueActuallyChanged: false,
    eventsDispatched: false,
    valueAfterEvents: '',
    rootCause: 'UNKNOWN',
    ok: false,
  };

  if (!candidateId) {
    diag.rootCause = 'DENOM_EMPTY';
    return { ok: false, error: 'DENOM_EMPTY: Нет candidateId', diag };
  }
  diag.candidateEntity = extractDenomEntity(candidateId);

  try {
    const dSel = win.document.getElementById('denomination-select');
    if (!dSel) {
      diag.rootCause = 'SELECT_NOT_FOUND';
      return { ok: false, error: 'DENOM_NO_SELECT: select не найден', diag };
    }
    diag.selectFound = true;
    diag.optionCount = dSel.options.length;
    /* Capture all options for diagnostic */
    for (const opt of dSel.options) {
      if (!opt.value) continue;
      const lbl = opt.textContent.trim();
      diag.options.push({
        value: opt.value,
        label: lbl,
        entity: extractDenomEntity(lbl),
      });
    }

    /* Current state */
    const currentOpt = dSel.options[dSel.selectedIndex];
    diag.currentRaw = currentOpt ? currentOpt.textContent.trim() : '';
    diag.currentEntity = extractDenomEntity(diag.currentRaw);
    diag.currentValue = dSel.value;
    diag.oldSelectValue = dSel.value;

    /* Check if current value already matches candidate canonically (MSG119) */
    if (currentOpt && denominationsCanonicallyEqual(currentOpt.textContent.trim(), candidateId)) {
      diag.alreadyCanonEqual = true;
      diag.rootCause = 'ALREADY_MATCHED';
      diag.ok = true;
      return { ok: true, value: currentOpt.value, label: norm(currentOpt.textContent),
               alreadyMatched: true, diag };
    }

    /* Entity-based matching: dn::TypeKey::value ↔ [TypeKey][mode][Cat][value] (MSG119) */
    const matched = findDenominationOptionByEntity(dSel, candidateId);
    if (!matched) {
      diag.rootCause = 'OPTION_NOT_FOUND';
      return { ok: false, error: 'DENOM_APPLY_NO_MATCH: Denomination кандидат «' + candidateId + '» не найден среди опций popup select', diag };
    }
    diag.matchedOptionValue = matched.value;
    diag.matchedOptionLabel = matched.textContent.trim();
    diag.matchedOptionEntity = extractDenomEntity(matched.textContent.trim());

    /* Apply: set value */
    dSel.value = matched.value;
    diag.newSelectValue = dSel.value;
    diag.valueActuallyChanged = (dSel.value === matched.value);

    if (!diag.valueActuallyChanged) {
      diag.rootCause = 'VALUE_SET_FAILED';
      return { ok: false, error: 'DENOM_VALUE_SET_FAILED: select.value не изменился после присвоения', diag };
    }

    /* Dispatch change/input events so page JS reacts */
    try {
      dSel.dispatchEvent(new Event('change', { bubbles: true }));
      dSel.dispatchEvent(new Event('input', { bubbles: true }));
      diag.eventsDispatched = true;
    } catch (_) { diag.eventsDispatched = false; }

    /* Check if value survived event handlers (no automatic revert) */
    diag.valueAfterEvents = dSel.value;
    if (dSel.value !== matched.value) {
      diag.rootCause = 'VALUE_REVERTED_AFTER_SET';
      return { ok: false, error: 'DENOM_VALUE_REVERTED: значение вернулось к «' + dSel.value + '» после dispatch events', diag };
    }

    diag.rootCause = 'SUCCESS';
    diag.ok = true;
    return { ok: true, value: matched.value, label: norm(matched.textContent), diag };
  } catch (e) {
    diag.rootCause = 'EXCEPTION: ' + (e.message || String(e));
    return { ok: false, error: 'DENOM_ERROR: ' + (e.message || e), diag };
  }
}

/* ── APPLY RECOMMENDATIONS ENGINE (Stage 1.3 — iterative chain) ─
   Iterative re-check: after changing ProductType → new backend call
   to get fresh Region recommendation; after changing Region → new
   backend call to get fresh Denomination recommendation.
   Each step uses REAL backend re-validation, not cached AI response. */
async function applyRecommendations(card) {
  const win = await ensureWorkerAtUrl(card.updateUrl);
  card.beforeSnapshot = readSnapshotFromPopup(win);
  card.applyError = null;   /* reset for fresh apply */

  const recs = card.recommendations;
  if (!recs) {
    card.applied = true;
    return;
  }

  const addError = note => {
    card.applyError = card.applyError ? card.applyError + '; ' + note : note;
  };

  let ptChanged = false;
  let rgChanged = false;

  /* ═══ STEP 1: ProductType ═══ */
  const ptStatus = (recs.productType || {}).status;
  if (ptStatus === 'mismatch' || ptStatus === 'uncertain') {
    const candId = (recs.productType || {}).candidateId || '';
    if (candId) {
      const res = applyProductTypeInPopup(win, candId);
      if (res.ok) {
        recs.productType.resolvedDomValue = res.value;
        recs.productType.resolvedDomLabel = res.label;
        ptChanged = true;
      } else {
        addError(res.error + '. Выберите ProductType вручную в popup.');
      }
    }
  }

  /* ═══ STEP 2: Re-check after ProductType change → fresh Region ═══ */
  let step2Recs = recs;   /* default: use original recs */
  if (ptChanged) {
    /* Re-read snapshot with new ProductType applied */
    const freshSnap = readSnapshotFromPopup(win) || card.beforeSnapshot;
    const payload2  = buildPayloadFromSnapshot(card, freshSnap);
    try {
      const raw2    = await callBackend(payload2);
      const mapped2 = mapBackendResult(raw2);
      step2Recs = mapped2.recommendations || recs;
      /* Update card's stored recommendations with fresh data */
      if (step2Recs.region) recs.region = step2Recs.region;
      if (step2Recs.denomination) recs.denomination = step2Recs.denomination;
      card.aiRaw = raw2;   /* keep latest raw */
    } catch (e) {
      addError('RECHECK_AFTER_PT: Повторный вызов backend после ProductType не удался: ' + (e.message || e));
    }
  }

  /* ═══ STEP 2b: Apply Region (MSG178/185: brand whitelist suppresses manual) ═══ */
  const rgApplyStatus = (recs.region || {}).status;
  if (rgApplyStatus === 'mismatch' || rgApplyStatus === 'uncertain') {
    const rgCand = (recs.region || {}).candidateId || '';
    if (rgCand) {
      /* MSG185: check brand whitelist before flagging as manual */
      const curSnap2b = readSnapshotFromPopup(win) || card.beforeSnapshot;
      const rgName2b = curSnap2b ? (curSnap2b.name || card.productName) : card.productName;
      const rgDesc2b = curSnap2b ? (curSnap2b.description || '') : '';
      const rgCurRegion2b = curSnap2b ? ((curSnap2b.regions || []).map(r => r.name).join(', ')) : '';
      const brandInf2b = checkBrandRegionInference(rgName2b, rgDesc2b);
      const brandConfirms2b = brandInf2b.matched &&
        (regionsCanonicallyEqual(rgCurRegion2b, brandInf2b.inferredRegion) ||
         regionsCanonicallyEqual(rgCand, brandInf2b.inferredRegion));
      if (brandConfirms2b) {
        /* Brand whitelist confirms region — no manual action needed */
        if (!recs.region) recs.region = {};
        recs.region.brandConfirmed = true;
        recs.region.brandFamily = brandInf2b.brand;
        recs.region.brandInferredRegion = brandInf2b.inferredRegion;
      } else {
        /* Region add/del requires server-side AJAX → still manual for now */
        addError('REGION_MANUAL: Регион требует ручной проверки в popup. ' +
          'Кандидат от AI: «' + rgCand + '».');
      }
    }
  }

  /* ═══ STEP 3: Re-check after Region → fresh Denomination ═══ */
  let step3Recs = recs;
  if (rgChanged || ptChanged) {
    /* Even if region was not auto-applied, re-check denomination based on current state */
    const freshSnap3 = readSnapshotFromPopup(win) || card.beforeSnapshot;
    const payload3   = buildPayloadFromSnapshot(card, freshSnap3);
    try {
      const raw3    = await callBackend(payload3);
      const mapped3 = mapBackendResult(raw3);
      step3Recs = mapped3.recommendations || recs;
      if (step3Recs.denomination) recs.denomination = step3Recs.denomination;
    } catch (e) {
      addError('RECHECK_AFTER_RG: Повторный вызов backend после Region не удался: ' + (e.message || e));
    }
  }

  /* ═══ STEP 3b: Apply Denomination (MSG123: candidate+canonical trigger) ═══ */
  /* Trigger rule: apply when candidate exists AND current differs canonically.
     Do NOT rely solely on module status — AI may return status=match while
     current and recommended are canonically different (MSG123 root cause). */
  const dCandId = (recs.denomination || {}).candidateId || '';
  let dnApplyTriggered = false;
  if (dCandId) {
    /* Read current denomination from popup for canonical comparison */
    const curSnap = readSnapshotFromPopup(win) || card.beforeSnapshot;
    const curDn = curSnap ? (curSnap.denominationLabel || curSnap.denominationOptionId || '') : '';
    const dnCanonEq = denominationsCanonicallyEqual(curDn, dCandId);
    if (dnCanonEq) {
      /* Already correct — mark as alreadyMatched, no apply needed */
      if (!recs.denomination) recs.denomination = {};
      recs.denomination.alreadyMatched = true;
      recs.denomination.applyDiag = {
        ts: isoNow(), candidateRaw: dCandId,
        candidateEntity: extractDenomEntity(dCandId),
        currentRaw: curDn, currentEntity: extractDenomEntity(curDn),
        alreadyCanonEqual: true, rootCause: 'ALREADY_MATCHED', ok: true,
        selectFound: true, optionCount: 0, options: [],
        matchedOptionValue: '', matchedOptionLabel: '', matchedOptionEntity: '',
        oldSelectValue: '', newSelectValue: '', valueActuallyChanged: false,
        eventsDispatched: false, valueAfterEvents: '', currentValue: '',
      };
    } else {
      /* Canonical mismatch — apply denomination */
      dnApplyTriggered = true;
      const res = applyDenominationInPopup(win, dCandId);
      if (res.diag) recs.denomination.applyDiag = res.diag;
      if (res.ok) {
        recs.denomination.resolvedDomValue = res.value;
        recs.denomination.resolvedDomLabel = res.label;
        if (res.alreadyMatched) {
          recs.denomination.alreadyMatched = true;
        }
      } else {
        addError(res.error);
      }
    }
  }

  card.afterSnapshot = readSnapshotFromPopup(win);
  /* MSG123 ACTION 2: applied=true only if something actually changed or was alreadyMatched */
  const ptApplied = (recs.productType || {}).resolvedDomValue != null;
  const dnApplied = (recs.denomination || {}).resolvedDomValue != null;
  const dnAlreadyOk = (recs.denomination || {}).alreadyMatched === true;
  const anyFieldApplied = ptChanged || ptApplied || dnApplied || dnAlreadyOk;
  card.applied = anyFieldApplied || (!card.applyError);
  try { win.focus(); } catch (_) {}
}

/* ── APPROVE / SAVE ENGINE ───────────────────────────────────── */
/* Controlled save: navigates popup to card URL, re-applies recommendations,
   then submits the form. No auto-save. Must be explicitly triggered. */
async function saveCard(card) {
  /* Navigate popup to the card page (re-apply ensures form state is correct) */
  await applyRecommendations(card);

  const win = _s.workerWin;
  if (!win || win.closed) throw new Error('WORKER_CLOSED: Popup недоступен для сохранения');

  const doc     = win.document;
  const saveBtn = doc.querySelector('input[name="yt0"]');
  const form    = doc.getElementById('product-form');
  if (!saveBtn && !form) throw new Error('SAVE_FORM_NOT_FOUND: Кнопка Save не найдена в popup');

  /* Navigation sentinel before submit */
  const seq = ++_s.navSeq;
  try { win.__bsv_nav_seq__ = seq; } catch (_) {}

  try {
    if (saveBtn) saveBtn.click();
    else form.submit();
  } catch (e) {
    throw new Error('SAVE_SUBMIT_ERROR: ' + e.message);
  }

  /* Wait for old page context destruction */
  const deadline = Date.now() + NAV_TIMEOUT_MS;
  while (Date.now() < deadline) {
    if (win.closed) throw new Error('WORKER_CLOSED: Popup закрыт после submit');
    try { if (win.__bsv_nav_seq__ !== seq) break; } catch (_) { break; }
    await wait(100);
  }
  await wait(300);

  /* Wait for new page (save redirects back to same page) */
  await waitForWorkerReady(win, PAGE_LOAD_MS);
  card.afterSnapshot = readSnapshotFromPopup(win);
  card.saved = true;
}

/* ── MOD STATUS ENGINE ───────────────────────────────────────── */
function getModStatusRule(color) {
  return MOD_STATUS_RULES.find(r => r.color === color) || null;
}

/* Try to extract Yii1 CSRF token (W5 — UNKNOWN in DOM snapshot).
   Yii1 embeds YII_CSRF_TOKEN as a meta tag or hidden input.
   If not found, sends without token (works if Yii CSRF is disabled for XHR). */
function extractCsrfToken() {
  try {
    const meta = document.querySelector('meta[name="csrf-token"]');
    if (meta) return meta.getAttribute('content') || '';
    const hidden = document.querySelector('input[name="YII_CSRF_TOKEN"]');
    if (hidden) return hidden.value || '';
  } catch (_) {}
  return '';
}

async function applyModStatus(card) {
  const rule = getModStatusRule(card.finalColor);
  if (!rule) throw new Error('MOD_STATUS_NO_RULE: Нет правила для цвета «' + card.finalColor + '»');
  if (rule._blocked) throw new Error('MOD_STATUS_BLOCKED: ' + rule._blockedReason);
  if (rule.modStatus === 'no_change') return;   /* skip silently */

  const csrfToken = extractCsrfToken();
  const bodyParts = [
    'product_id='    + encodeURIComponent(card.productId),
    'mod_status='    + encodeURIComponent(rule.modStatus),
    'rejection_note=' + encodeURIComponent(rule.rejectionNote || ''),
  ];
  if (rule.requiresRejectionReason && rule.rejectionReasonId) {
    bodyParts.push('rejection_reason=' + encodeURIComponent(rule.rejectionReasonId));
  }
  if (csrfToken) bodyParts.push('YII_CSRF_TOKEN=' + encodeURIComponent(csrfToken));
  const body = bodyParts.join('&');

  const resp = await new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open('POST', '/xdm/product/updateModStatus', true);
    xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
    xhr.setRequestHeader('X-Requested-With', 'XMLHttpRequest');
    if (csrfToken) xhr.setRequestHeader('X-CSRF-Token', csrfToken);
    xhr.timeout = 15000;
    xhr.onload  = () => {
      try { resolve(JSON.parse(xhr.responseText)); }
      catch (_) { resolve({ success: false, error: 'Parse error: ' + xhr.responseText.slice(0, 100) }); }
    };
    xhr.onerror   = () => reject(new Error('XHR network error'));
    xhr.ontimeout = () => reject(new Error('XHR timeout 15s'));
    xhr.send(body);
  });

  if (!resp || !resp.success) {
    throw new Error('MOD_STATUS_FAIL: ' + ((resp && resp.error) ? resp.error : 'Сервер вернул ошибку'));
  }

  card.modStatusApplied = true;
  card.modStatusNew     = rule.modStatus;

  /* Update live dropdown on list page */
  try {
    const sel = document.querySelector(
      'select.mod-status-dropdown[data-id="' + card.productId + '"],' +
      '#mod_status_' + card.productId
    );
    if (sel) sel.value = rule.modStatus;
  } catch (_) {}
}

/* ── EXPORT ENGINE ───────────────────────────────────────────── */
/* BACKEND GAP: Apps Script→Sheets export requires v16.gs (new writeExportRows action).
   Stage 1 interim: clipboard TSV.                                                      */
function buildExportTsv(session, cards) {
  const hdr = [
    'sessionId', 'scannedAt', 'productId', 'productName',
    'ptBefore', 'ptAfter', 'regionsBefore', 'regionsAfter',
    'denomBefore', 'denomAfter',
    'aiVerdict', 'aiConfidence', 'aiReason',
    'operatorOverride', 'finalColor',
    'applied', 'saved', 'manuallyConfirmed', 'modStatusChanged', 'modStatusNew',
    'scanError', 'applyError', 'saveError',
  ].join('\t');

  const snap2pt  = s => s ? (s.productTypeName || s.productTypeId || '—') : '—';
  const snap2reg = s => s ? ((s.regions || []).map(r => r.name).join(', ') || '—') : '—';
  const snap2den = s => s ? (s.denominationLabel || s.denominationOptionId || '—') : '—';

  const rows = cards.map(c => [
    session.sessionId,          session.scannedAt,
    c.productId,                c.productName,
    snap2pt(c.beforeSnapshot),  snap2pt(c.afterSnapshot),
    snap2reg(c.beforeSnapshot), snap2reg(c.afterSnapshot),
    snap2den(c.beforeSnapshot), snap2den(c.afterSnapshot),
    c.aiVerdict    || '—',
    c.aiConfidence != null ? c.aiConfidence : '—',
    c.aiReason     || '—',
    c.operatorOverride || '—',  c.finalColor || '—',
    c.applied  ? 'yes' : 'no', c.saved ? 'yes' : 'no',
    c.manuallyConfirmed ? 'yes' : 'no',
    c.modStatusApplied ? 'yes' : 'no', c.modStatusNew || '—',
    c.scanError  || '—',
    c.applyError || '—',
    c.saveError  || '—',
  ].map(v => String(v).replace(/\t|\n/g, ' ')).join('\t'));

  return [hdr, ...rows].join('\n');
}

async function runExport(session, cards) {
  if (!cards.length) { alert('Нет карточек для экспорта.'); return; }
  const tsv = buildExportTsv(session, cards);
  let method = 'clipboard';
  try {
    await navigator.clipboard.writeText(tsv);
  } catch (_) {
    /* Fallback: open in new tab */
    method = 'window';
    const w = window.open('', '_blank');
    if (w) { w.document.write('<pre style="font:12px monospace;padding:16px">' + esc(tsv) + '</pre>'); }
  }
  if (EXPORT_CONFIG.backendGap) {
    alert(
      'Экспорт: ' + cards.length + ' строк → ' + method + '\n\n' +
      'BACKEND GAP: Экспорт в Google Sheets (таб «' + EXPORT_CONFIG.exportTab + '») ' +
      'недоступен в Stage 1.\n' +
      'Требуется v16.gs с action=writeExportRows.'
    );
  } else {
    alert('Экспорт: ' + cards.length + ' строк → ' + method);
  }
}

/* ── EXPORT SERVICE CALLER ───────────────────────────────────── */
/* Calls LIST_CHECK_EXPORT_SERVICE_v1 via postmessage iframe transport.
   Same pattern as callBackend but different source identifier. */
function callExportService(payload) {
  if (!EXPORT_CONFIG.exportServiceUrl) return Promise.resolve(null);
  return new Promise(resolve => {
    const reqId  = 'exp_' + Date.now() + '_' + Math.random().toString(36).slice(2);
    const iframe = document.createElement('iframe');
    const form   = document.createElement('form');
    iframe.name       = 'exp_f_' + reqId;
    iframe.style.display = 'none';
    form.method       = 'POST';
    form.action       = EXPORT_CONFIG.exportServiceUrl;
    form.target       = iframe.name;
    form.style.display = 'none';
    const addInput = (n, v) => {
      const i = document.createElement('input');
      i.type = 'hidden'; i.name = n; i.value = String(v ?? '');
      form.appendChild(i);
    };
    addInput('transport',         'postmessage');
    addInput('client_request_id', reqId);
    const enriched = Object.assign({}, payload || {});
    if (EXPORT_CONFIG.exportAuthToken) enriched.auth_token = EXPORT_CONFIG.exportAuthToken;
    addInput('payload_json',      JSON.stringify(enriched));
    let settled = false;
    const tId = setTimeout(() => {
      if (settled) return; settled = true; cleanup();
      resolve({ ok: false, error: 'Export service timeout' });
    }, EXPORT_CONFIG.serviceTimeout);
    function cleanup() {
      const fn = _s.msgHandlers.get(reqId);
      if (fn) { window.removeEventListener('message', fn); _s.msgHandlers.delete(reqId); }
      clearTimeout(tId);
      setTimeout(() => { try { form.remove(); } catch(_){} try { iframe.remove(); } catch(_){} }, 600);
    }
    const handler = ev => {
      const d = ev && ev.data;
      if (!d || typeof d !== 'object') return;
      if (d.source !== EXPORT_CONFIG.exportServiceSrc) return;
      if (String(d.client_request_id || '') !== reqId) return;
      if (settled) return;
      settled = true; cleanup(); resolve(d);
    };
    _s.msgHandlers.set(reqId, handler);
    window.addEventListener('message', handler);
    document.body.appendChild(iframe);
    document.body.appendChild(form);
    try { form.submit(); } catch (e) { cleanup(); resolve({ ok: false, error: e.message }); }
  });
}

  const snap2reg = s => s ? ((s.regions || []).map(r => r.name).join(', ') || '') : '';

  return cards.map(c => ({

    productId:         c.productId,
    productName:       c.productName,
    ptBefore:          snap2pt(c.beforeSnapshot),
    ptAfter:           snap2pt(c.afterSnapshot),
    regionsBefore:     snap2reg(c.beforeSnapshot),
    regionsAfter:      snap2reg(c.afterSnapshot),
    denomBefore:       snap2den(c.beforeSnapshot),
    denomAfter:        snap2den(c.afterSnapshot),
    aiVerdict:         c.aiVerdict || '',
    aiConfidence:      c.aiConfidence != null ? String(c.aiConfidence) : '',
    aiReason:          c.aiReason || '',
    operatorOverride:  c.operatorOverride || '',
    finalColor:        c.finalColor || '',
    applied:           c.applied ? 'yes' : 'no',
    saved:             c.saved ? 'yes' : 'no',
    manuallyConfirmed: c.manuallyConfirmed ? 'yes' : 'no',

    modStatusNew:      c.modStatusNew || '',
    scanError:         c.scanError || '',
    applyError:        c.applyError || '',

  }));
}

async function exportToSheet(session, cards) {
  if (!cards.length) { alert('Нет карточек для отправки в таблицу.'); return; }

  const rows = buildExportRows(session, cards);

  /* Try export service first */
  if (EXPORT_CONFIG.exportServiceUrl) {
    const resp = await callExportService({ action: 'writeExportRows', rows: rows });
    if (resp && resp.ok) {
      cards.forEach(c => { c.exportedToSheet = true; });
      alert('В таблицу: ' + resp.written + ' строк записано в Google Sheet.');
      return;
    }
    /* Service call failed — fall through to clipboard */
    const errMsg = (resp && resp.error) ? resp.error : 'Нет ответа';
    alert('Export service ошибка: ' + errMsg + '\nИспользую clipboard fallback.');
  }

  /* Fallback: clipboard + open sheet */
  const tsv = buildExportTsv(session, cards);
  try { await navigator.clipboard.writeText(tsv); } catch (_) {}
  window.open(EXPORT_CONFIG.sheetUrl, '_blank');
  cards.forEach(c => { c.exportedToSheet = true; });
  alert(
    'В таблицу: ' + cards.length + ' строк скопировано в буфер обмена.\n' +
    'Google Sheet открыт — вставьте данные (Ctrl+V).\n\n' +
    (EXPORT_CONFIG.exportServiceUrl
      ? 'Export service вернул ошибку — использован clipboard fallback.'
      : 'Export service не настроен (exportServiceUrl пуст). Установите URL после деплоя LIST_CHECK_EXPORT_SERVICE_v1.gs.')
  );
}

/* ── SCAN LOOP ───────────────────────────────────────────────── */
async function runScan(items, session, onProgress) {
  _s.isFirstCard = true;

  for (let i = 0; i < items.length; i++) {
    if (_s.aborted) break;
    const item = items[i];
    const card = makeCardResult(item.productId, item.productName, item.productType, item.updateUrl);
    session.cards.push(card);

    try {
      /* 1. Open popup → capture beforeSnapshot */
      const win = await ensureWorkerAtUrl(item.updateUrl);
      card.beforeSnapshot = readSnapshotFromPopup(win);

      /* 2. Build payload from snapshot */
      const snap = card.beforeSnapshot || makeSnap();
      const payload = {
        product_id:   item.productId,
        name:         snap.name        || item.productName,
        description:  snap.description || '',
        current: {
          product_type: snap.productTypeName || item.productType,
          region:       (snap.regions || []).map(r => r.name).join(', '),
          denomination: snap.denominationLabel || '',
        },
        mode: 'card',
      };

      /* 3. Call AI Core v1 backend (postmessage from list page window) */
      const raw    = await callBackend(payload);
      const mapped = mapBackendResult(raw);
      card.aiVerdict       = mapped.verdict;
      card.aiConfidence    = mapped.confidence;
      card.aiReason        = mapped.reason;
      card.aiRaw           = raw;
      card.recommendations = mapped.recommendations;

    } catch (e) {
      card.aiVerdict = 'error';
      card.scanError = e.message || String(e);
      if (e.message && e.message.includes('POPUP_BLOCKED')) { _s.aborted = true; break; }
    }

    /* Update summary */
    const v = card.aiVerdict;
    session.summary.processed++;
    if (v === 'green')  session.summary.green++;
    else if (v === 'yellow') session.summary.yellow++;
    else if (v === 'red')    session.summary.red++;
    else                     session.summary.errors++;

    onProgress(i + 1, items.length, card);
    await wait(60);   /* breathing room */
  }

  /* Close worker after scan completes (will reopen for apply) */
  if (_s.workerWin && !_s.workerWin.closed) {
    try { _s.workerWin.close(); } catch (_) {}
    _s.workerWin = null;
    _s.isFirstCard = true;
  }
}

}

/* ── STARTUP (MSG129: card selection UI) ────────────────────── */
function renderStartup(items, onStartWithItems) {
  _s.uiPhase = 'startup';
  const count = items.length;
  /* Track manual checkbox state per productId */
  const checked = new Set();
  let manualMode = false; /* true once operator toggles any checkbox */

  function getEffectiveCount() {
    if (manualMode && checked.size > 0) return checked.size;
    const inp = $('wip01-scan-count');
    let n = inp ? parseInt(inp.value, 10) : count;
    if (isNaN(n) || n < 1) n = 1;
    if (n > count) n = count;
    return n;
  }

  function updateStartButton() {
    const btn = $('wip01-start');
    if (!btn) return;
    const eff = getEffectiveCount();
    const mode = (manualMode && checked.size > 0) ? ' выбрано' : '';
    btn.textContent = 'Начать проверку (' + eff + ' карточ' + pluralCard(eff) + mode + ')';
  }

  function pluralCard(n) {
    const m = n % 10, h = n % 100;
    if (m === 1 && h !== 11) return 'ка';
    if (m >= 2 && m <= 4 && (h < 12 || h > 14)) return 'ки';
    return 'ек';
  }

  function updateCounter() {
    const el = $('wip01-sel-counter');
    if (!el) return;
    if (manualMode && checked.size > 0) {
      el.textContent = 'Выбрано вручную: ' + checked.size + ' из ' + count;
      el.style.color = '#60a5fa';
    } else {
      el.textContent = '';
    }
  }

  function toggleCheck(pid) {
    manualMode = true;
    if (checked.has(pid)) checked.delete(pid); else checked.add(pid);
    /* Update checkbox visual */
    const cb = $('wip01-cb-' + pid);
    if (cb) cb.checked = checked.has(pid);
    updateStartButton();
    updateCounter();
  }

  function setAll(on) {
    manualMode = true;
    checked.clear();
    if (on) items.forEach(it => checked.add(it.productId));
    for (const it of items) {
      const cb = $('wip01-cb-' + it.productId);
      if (cb) cb.checked = on;
    }
    updateStartButton();
    updateCounter();
  }

  /* Build card list HTML */
  let listHtml = '';
  for (const it of items) {
    listHtml += `<label id="wip01-row-${esc(it.productId)}" style="display:flex;align-items:center;gap:8px;
      padding:7px 10px;border-radius:8px;cursor:pointer;transition:background .15s;
      border-bottom:1px solid rgba(255,255,255,.04);"
      onmouseover="this.style.background='rgba(96,165,250,.08)'"
      onmouseout="this.style.background='transparent'">
      <input type="checkbox" id="wip01-cb-${esc(it.productId)}" data-pid="${esc(it.productId)}"
        style="width:16px;height:16px;accent-color:#2563eb;cursor:pointer;flex-shrink:0;">
      <span style="font-size:11px;color:#6b7280;min-width:52px;">#${esc(it.productId)}</span>
      <span style="font-size:12px;color:#e5e7eb;flex:1;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${esc(it.productName)}</span>
      <span style="font-size:10px;color:#64748b;max-width:100px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${esc(it.productType)}</span>
    </label>`;
  }

  $('wip01-list').innerHTML = `
    <div style="padding:14px 14px 0;">
      <div style="font-size:13px;color:#94a3b8;text-align:center;">Товаров <b style="color:#a5b4fc;">Under Review</b> на странице:</div>
      <div style="font-size:42px;font-weight:800;color:#60a5fa;margin:6px 0 12px;text-align:center;">${count}</div>
      ${count > 0 ? `
        <div style="margin-bottom:10px;text-align:center;">
          <label style="font-size:13px;color:#94a3b8;">Сканировать первых:</label>
          <input id="wip01-scan-count" type="number" min="1" max="${count}" value="${count}"
            style="width:70px;margin-left:8px;padding:6px 10px;border-radius:8px;border:1px solid rgba(255,255,255,.2);
            background:rgba(255,255,255,.07);color:#e5e7eb;font-size:16px;font-weight:700;text-align:center;">
          <span style="font-size:12px;color:#6b7280;margin-left:4px;">из ${count}</span>
        </div>
        <div style="display:flex;gap:8px;margin-bottom:8px;justify-content:center;">
          <button id="wip01-sel-all" type="button" style="border:1px solid rgba(255,255,255,.15);
            border-radius:6px;padding:4px 12px;background:rgba(255,255,255,.05);color:#94a3b8;
            font-size:11px;cursor:pointer;">Выбрать все</button>
          <button id="wip01-sel-none" type="button" style="border:1px solid rgba(255,255,255,.15);
            border-radius:6px;padding:4px 12px;background:rgba(255,255,255,.05);color:#94a3b8;
            font-size:11px;cursor:pointer;">Снять все</button>
        </div>
        <div id="wip01-sel-counter" style="font-size:12px;text-align:center;margin-bottom:8px;min-height:16px;"></div>
      ` : ''}
    </div>
    ${count > 0 ? `
      <div style="flex:1;overflow-y:auto;padding:0 8px 8px;max-height:calc(100vh - 360px);">
        ${listHtml}
      </div>
      <div style="padding:10px 14px;">
        <button id="wip01-start" type="button" style="border:none;border-radius:10px;padding:12px 24px;
          background:#2563eb;color:#fff;font-weight:700;font-size:15px;cursor:pointer;width:100%;">
          Начать проверку (${count} карточ${pluralCard(count)})</button>
      </div>
    ` : '<div style="padding:14px;color:#f59e0b;text-align:center;">Нет товаров Under Review.</div>'}`;

  $('wip01-detail').innerHTML = '<div style="color:#6b7280;padding:24px;text-align:center;">Выберите карточки для проверки или нажмите «Начать проверку».</div>';

  /* Bind checkbox events */
  for (const it of items) {
    const cb = $('wip01-cb-' + it.productId);
    if (cb) cb.onchange = () => toggleCheck(it.productId);
  }

  /* Bind select all / deselect all */
  const selAll = $('wip01-sel-all');
  if (selAll) selAll.onclick = () => setAll(true);
  const selNone = $('wip01-sel-none');
  if (selNone) selNone.onclick = () => setAll(false);

  /* Update button text when count input changes */
  const scanInp = $('wip01-scan-count');
  if (scanInp) scanInp.oninput = () => {
    if (!manualMode || checked.size === 0) updateStartButton();
  };

  /* Bind start button */
  const startBtn = $('wip01-start');
  if (startBtn) startBtn.onclick = () => {
    let selectedItems;
    if (manualMode && checked.size > 0) {
      /* Manual selection mode: only checked items, in page order */
      selectedItems = items.filter(it => checked.has(it.productId));
    } else {
      /* Count-based mode: first N items */
      let n = scanInp ? parseInt(scanInp.value, 10) : count;
      if (isNaN(n) || n < 1) n = 1;
      if (n > count) n = count;
      selectedItems = items.slice(0, n);
    }
    onStartWithItems(selectedItems);
  };
}

/* ── TICKET CONTEXT BLOCK ───────────────────────────────────── */
function renderTicketContext(card) {
  const b   = card.beforeSnapshot;
  const raw = card.aiRaw || {};
  const sel = raw.selection || {};
  const pt  = b ? (b.productTypeName || b.productTypeId || '—') : card.productType || '—';
  const rg  = b ? ((b.regions || []).map(r => r.name).join(', ') || '—') : '—';
  const dn  = b ? (b.denominationLabel || b.denominationOptionId || '—') : '—';
  const nm  = b ? (b.name || card.productName) : card.productName;
  const desc = b ? (b.description || '') : '';

  /* Replacement hint builder (unified pattern — MSG81) */
  const hintBox = lines => `<div style="margin-top:3px;padding:4px 8px;background:#fef3c7;border:1px solid #f59e0b;border-radius:4px;font-size:12px;color:#92400e;">${lines}</div>`;

  /* ── Candidate comparisons (MUST be declared before evidence terms use them — MSG95 fix) ── */
  const ptCand = sel.product_type_candidate_id || '';
  const ptModStatus = ((raw.modules || {}).product_type || {}).status || '';
  /* MSG137: Catalogue-driven ProductType resolution */
  const ptOptions = b ? (b.productTypeOptions || []) : [];
  const ptResolved = resolveProductTypeState(pt, ptCand, ptOptions, ptModStatus, nm);
  const ptCanonMatch = ptResolved.semanticMatch;
  const rgCand = sel.region_candidate_id || '';
  const rgMatch = regionsCanonicallyEqual(rg, rgCand);
  const dnCandTc = sel.denomination_candidate_id || '';
  const dnMatchTc = !dnCandTc || denominationsCanonicallyEqual(dn, dnCandTc);

  /* ProductType hint — MSG137/142: use resolved display label, handle suspect candidates */
  let ptHelper = '';
  if (ptResolved.showHelper && ptResolved.helperMode === 'replacement') {
    ptHelper = hintBox(
      `Название/описание указывает на другой ProductType: <b>${esc(ptResolved.recommendedDisplay)}</b><br>` +
      `В таблице есть ProductType: <b>${esc(ptResolved.recommendedDisplay)}</b>`);
  } else if (ptResolved.showHelper && ptResolved.helperMode === 'unresolved') {
    /* MSG142: suspect candidate — warn operator, don't suggest replacement */
    ptHelper = hintBox(
      `AI предложил ProductType: <b>${esc(ptResolved.recommendedDisplay)}</b>, но это может быть ложное совпадение по platform keyword.<br>` +
      `Требуется ручная проверка ProductType.`);
  }

  /* Region hint (MSG135 + MSG178: brand inference whitelist) */
  const rgModTc = (raw.modules || {}).region || {};
  const rgUncertainTc = rgModTc.status === 'uncertain';
  const rgBrandInfTc = checkBrandRegionInference(nm, desc);
  const rgBrandConfirmsTc = rgBrandInfTc.matched &&
    (regionsCanonicallyEqual(rg, rgBrandInfTc.inferredRegion) ||
     regionsCanonicallyEqual(rgCand, rgBrandInfTc.inferredRegion));
  let rgHelper = '';
  if (rgCand && !rgMatch && !rgBrandConfirmsTc) {
    /* Replacement case: canonical mismatch, no brand inference */
    rgHelper = hintBox(
      `В name найден другой регион: <b>${esc(rgCand)}</b><br>` +
      `В таблице есть регион: <b>${esc(rgCand)}</b>`);
  } else if (rgCand && rgMatch && rgUncertainTc && !rgBrandConfirmsTc) {
    /* Evidence-warning case: canonical match but no text evidence and no brand inference (MSG135) */
    rgHelper = hintBox(
      `Регион совпадает с candidate (canonical match), но seller text не подтверждает. Требуется проверка.`);
  }
  /* MSG178: brand inference does NOT show replacement/warning helper — it's a green-path */

  /* Denomination hint (MSG135 + MSG178: brand inference unblocks region) */
  const dnModTc = (raw.modules || {}).denomination || {};
  const dnUncertainTc = dnModTc.status === 'uncertain';
  const rgStillUncertainTc = rgUncertainTc && !rgBrandConfirmsTc;
  let dnHelper = '';
  if (dnCandTc && !dnMatchTc) {
    /* Replacement case */
    dnHelper = hintBox(
      `В name/description найден другой denomination: <b>${esc(dnCandTc)}</b><br>` +
      `В таблице есть denomination: <b>${esc(dnCandTc)}</b>`);
  } else if (dnCandTc && dnMatchTc && (dnUncertainTc || rgStillUncertainTc)) {
    /* Evidence-warning case: canonical match but evidence missing or region unconfirmed (MSG135) */
    const reason = rgStillUncertainTc ? 'регион не подтверждён seller text' : 'text evidence отсутствует';
    dnHelper = hintBox(
      `Denomination совпадает с candidate, но ${reason}. Автоподтверждение заблокировано.`);
  }

  /* Evidence highlighting (MSG89/97/103): collect terms that influenced the decision */
  /* MSG158: PT aliases from snapshot, with embedded fallback list */
  const PT_ALIASES_EMBEDDED = ['itunes','apple','psn','playstation network','lol','pubg','pubg mobile',
    'razer','razer pin','currys','google play','spotify','fortnite','xbox','steam','nintendo',
    'gift card','gift cards','subscription','game key','game keys'];
  const PT_ALIASES = _aliasState.loaded && _aliasState.ptAliases.length > 0
    ? [...new Set([..._aliasState.ptAliases, ...PT_ALIASES_EMBEDDED])]
    : PT_ALIASES_EMBEDDED;
  const evidenceTerms = [];   /* [{term, type:'confirm'|'replace'}] */
  const addEvidence = (term, type) => {
    if (!term || term === '—' || term.length < 2) return;
    if (evidenceTerms.some(e => e.term.toLowerCase() === term.toLowerCase())) return;
    evidenceTerms.push({ term, type });
  };
  /* Helper: split compound PT name into individual meaningful words */
  const splitPtTerms = (s) => {
    if (!s) return [];
    const terms = [];
    /* Split on separators first */
    s.split(/[-–_/,]/).forEach(chunk => {
      const t = chunk.trim();
      if (t.length >= 3) terms.push(t);
      /* Also split multi-word chunks into individual words ≥3 chars */
      t.split(/\s+/).forEach(w => { if (w.length >= 3 && w !== t) terms.push(w); });
    });
    return terms;
  };
  const ptType = ptCanonMatch ? 'confirm' : 'replace';
  /* PT terms: full name + parts + individual words.
     MSG137: use resolved display label for evidence, not raw slug */
  const ptDisplayForEvidence = ptResolved.recommendedDisplay || ptCand;
  if (pt && pt !== '—') {
    addEvidence(pt, 'confirm');
    splitPtTerms(pt).forEach(t => addEvidence(t, 'confirm'));
  }
  if (ptDisplayForEvidence) splitPtTerms(ptDisplayForEvidence).forEach(t => addEvidence(t, ptType));
  /* PT aliases: check which known aliases appear to be relevant */
  PT_ALIASES.forEach(alias => {
    const al = alias.toLowerCase();
    if ((pt && pt.toLowerCase().includes(al)) || (ptCand && ptCand.toLowerCase().includes(al))) {
      addEvidence(alias, ptType);
    }
  });
  /* Region (MSG132: canonical match → confirm, not replace) */
  if (rg && rg !== '—') rg.split(',').forEach(r => addEvidence(r.trim(), 'confirm'));
  if (rgCand && rgCand.length < 30) addEvidence(rgCand, rgMatch ? 'confirm' : 'replace');
  /* Denomination */
  if (dn && dn !== '—' && dn.length < 60) addEvidence(dn, 'confirm');
  if (dnCandTc && !dnMatchTc && dnCandTc.length < 60) addEvidence(dnCandTc, 'replace');
  if (dnCandTc) {
    const dnParts = dnCandTc.match(/^dn::(.+?)::(.+)$/i);
    if (dnParts) addEvidence(dnParts[2], dnMatchTc ? 'confirm' : 'replace');
  }

  /* Build highlight function — single-pass to avoid nested HTML corruption (MSG103 fix) */
  function highlightEvidence(text) {
    if (!text || !evidenceTerms.length) return esc(text);
    /* Work on raw text, not pre-escaped */
    const raw = String(text);
    /* Build combined regex from all terms, longest first */
    const sorted = [...evidenceTerms].sort((a, b) => b.term.length - a.term.length);
    const patterns = sorted.map(ev => ev.term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'));
    if (!patterns.length) return esc(text);
    const combinedRe = new RegExp('(' + patterns.join('|') + ')', 'gi');
    /* Split text into parts: matched vs unmatched */
    const parts = raw.split(combinedRe);
    return parts.map(part => {
      const ev = sorted.find(e => e.term.toLowerCase() === part.toLowerCase());
      if (ev) {
        const style = ev.type === 'confirm'
          ? 'background:#dcfce7;color:#14532d;padding:0 3px;border-radius:3px;font-weight:600'
          : 'background:#fef3c7;color:#92400e;padding:0 3px;border-radius:3px;font-weight:600';
        return '<span style="' + style + '">' + esc(part) + '</span>';
      }
      return esc(part);
    }).join('');
  }

  const toneStyles = {
    neutral: 'border:1px solid #d1d5db;background:#fff;color:#1e293b;',
    green:   'border:1px solid #86efac;background:#f0fdf4;color:#14532d;',
    yellow:  'border:1px solid #fcd34d;background:#fefce8;color:#713f12;',
    red:     'border:1px solid #fca5a5;background:#fef2f2;color:#7f1d1d;',
  };
  const field = (label, value, tone, helper) => {
    const ts = toneStyles[tone] || toneStyles.neutral;
    return `<div style="display:grid;grid-template-columns:160px 1fr;border-bottom:1px solid #e5e7eb;">
      <div style="padding:8px 12px;font-weight:700;font-size:14px;color:#475569;border-right:1px solid #e5e7eb;">${esc(label)}</div>
      <div style="padding:8px 12px;">
        <div style="padding:7px 10px;font-size:14px;border-radius:4px;${ts}">${esc(value)}</div>
        ${helper || ''}
      </div>
    </div>`;
  };
  /* Dropdown field — shows real options from popup snapshot (MSG97 + MSG100) */
  const selectField = (label, currentVal, options, recommended, canonMatch, tone, helper) => {
    const ts = toneStyles[tone] || toneStyles.neutral;
    if (!options || !options.length) return field(label, currentVal, tone, helper);
    const showRec = recommended && !canonMatch;   /* hide recommendation when canonical match */
    const opts = options.map(o => {
      const isCurrent = (o.value === currentVal || o.label === currentVal);
      const isRec = showRec && (o.label.toLowerCase().includes(recommended.toLowerCase()) ||
        o.value.toLowerCase().includes(recommended.toLowerCase()));

function renderDetailPanel(session) {
  const detail = $('wip01-detail');
  if (!detail) return;

  /* Phase guard: don't render results if we're still in startup/scanning */
  if (_s.uiPhase === 'scanning') return;   /* progress handler owns right panel */
  if (_s.uiPhase === 'startup') {
    detail.innerHTML = '<div style="color:#6b7280;padding:24px;text-align:center;">Нажмите «Начать проверку» для запуска.</div>';
    return;
  }
  if (_s.uiPhase === 'empty') {
    detail.innerHTML = '<div style="color:#6b7280;padding:24px;text-align:center;">Нет результатов для отображения.</div>';
    return;
  }

  /* Phase = results: find active card with fallback to first */
  let card = session.cards.find(c => c.productId === _s.uiSelectedId);
  if (!card && session.cards.length > 0) {
    card = session.cards[0];
    _s.uiSelectedId = card.productId;
  }
  if (!card) {
    _s.uiPhase = 'empty';
    detail.innerHTML = '<div style="color:#6b7280;padding:24px;text-align:center;">Нет результатов для отображения.</div>';
    return;
  }
  const c   = clr(card.finalColor);
  const raw = card.aiRaw || {};
  const mods = raw.modules || {};
  const sel  = raw.selection || {};
  const recs = card.recommendations || {};
  const b = card.beforeSnapshot, a = card.afterSnapshot;
  const pt2  = s => s ? (s.productTypeName || s.productTypeId || '—') : '—';
  const reg2 = s => s ? ((s.regions || []).map(r => r.name).join(', ') || '—') : '—';
  const den2 = s => s ? (s.denominationLabel || s.denominationOptionId || '—') : '—';

  /* Ticket Context form block (above analysis) */
  const ticketCtx = renderTicketContext(card);

  /* ── Compute child block statuses FIRST, then aggregate for Block 1 ── */

  /* Block 2: Manual Review Words */
  const mrMod = mods.manual_review || {};
  const mrStatus = mrMod.status === 'match' ? 'red' : 'green';
  const sec2 = renderDetailSection('Manual Review Words', 2, mrStatus, [
    { label: 'Статус', value: mrMod.status === 'match' ? 'Найдены стоп-слова' : 'Стоп-слова не найдены.' },
    { label: 'Источник', value: 'manual_review_words' },
    { label: 'Совпадения', value: mrMod.reason || 'не найдено' },
  ]);

  /* Block 3: ProductType (MSG84 + MSG137 catalogue-driven) */
  const ptMod = mods.product_type || {};
  const ptCand3 = sel.product_type_candidate_id || '';
  const ptCur3 = pt2(b);
  const ptOpts3 = b ? (b.productTypeOptions || []) : [];
  const ptName3 = b ? (b.name || card.productName) : card.productName;
  const ptRes3 = resolveProductTypeState(ptCur3, ptCand3, ptOpts3, ptMod.status, ptName3);
  const ptStatus = ptRes3.uiStatus;
  const ptMatchLabel = ptRes3.semanticMatch
    ? (ptRes3.catalogueResolved ? '(catalogue match)' : '(canonical match)')
    : '';
  let ptAction3;
  if (ptRes3.semanticMatch) {
    ptAction3 = `ProductType подтверждён ${ptMatchLabel}.`;
  } else if (ptRes3.candidateSuspect) {
    ptAction3 = 'AI candidate может быть ложным (platform keyword). Требуется ручная проверка.';
  } else {
    ptAction3 = ptMod.reason || '—';
  }
  const sec3 = renderDetailSection('ProductType', 3, ptStatus, [
    { label: 'Текущий', value: ptCur3 },
    { label: 'Рекомендованный', html: esc(ptRes3.recommendedDisplay || ptCur3) +
      (ptRes3.semanticMatch && ptCand3 ? ` <span style="color:#4ade80;font-size:11px;">${ptMatchLabel}</span>` : '') +
      (ptRes3.candidateSuspect ? ' <span style="color:#f59e0b;font-size:11px;">(suspect — platform keyword)</span>' : '') },
    { label: 'Действие', value: ptAction3 },
    { label: 'Источник', value: 'catalogue' },
  ]);

  /* Block 4: Регион (MSG70 + MSG135 + MSG178: brand inference whitelist) */
  const rgMod = mods.region || {};
  const rgCand2 = sel.region_candidate_id || '';
  const rgCurrent = reg2(b);
  const rgCanonMatch = regionsCanonicallyEqual(rgCurrent, rgCand2);
  const rgTextConfirmed = rgMod.status === 'match';
  const rgUncertain = rgMod.status === 'uncertain';
  /* MSG178: brand → region inference check */
  const rgProductName = b ? (b.name || card.productName) : card.productName;
  const rgProductDesc = b ? (b.description || '') : '';
  const rgBrandInf = checkBrandRegionInference(rgProductName, rgProductDesc);
  /* Brand inference is valid when: brand matches, inferred region matches candidate or current,
     and there is no explicit contradicting region signal from another country */
  const rgBrandConfirms = rgBrandInf.matched &&
    (regionsCanonicallyEqual(rgCurrent, rgBrandInf.inferredRegion) ||
     regionsCanonicallyEqual(rgCand2, rgBrandInf.inferredRegion));
  let rgStatus;
  let rgSource = 'uncertain';  /* explicit_text | brand_region_whitelist | canonical_alias | uncertain */
  if (rgCanonMatch && rgTextConfirmed) {
    rgStatus = 'green'; rgSource = 'explicit_text';
  } else if (rgBrandConfirms && rgCanonMatch) {
    rgStatus = 'green'; rgSource = 'brand_region_whitelist';
  } else if (rgBrandConfirms && rgUncertain) {
    rgStatus = 'green'; rgSource = 'brand_region_whitelist';
  } else if (rgCanonMatch && rgUncertain) {
    rgStatus = 'yellow'; rgSource = 'canonical_alias';
  } else if (rgCanonMatch && !rgUncertain) {
    rgStatus = 'green'; rgSource = 'canonical_alias';
  } else if (rgTextConfirmed) {
    rgStatus = 'green'; rgSource = 'explicit_text';
  } else if (rgMod.status === 'mismatch') {
    rgStatus = 'red';
  } else {
    rgStatus = 'yellow';
  }
  /* Hints */
  let rgHints = '';
  if (rgCand2 && !rgCanonMatch && !rgBrandConfirms) {
    rgHints = `<div style="margin-top:4px;font-size:12px;color:#fbbf24;">В name найден другой регион: <b>${esc(rgCand2)}</b></div>` +
              `<div style="font-size:12px;color:#fbbf24;">В таблице есть регион: <b>${esc(rgCand2)}</b></div>`;
  } else if (rgCanonMatch && rgUncertain && !rgBrandConfirms) {
    rgHints = `<div style="margin-top:4px;font-size:12px;color:#fbbf24;">Canonical match, но seller text не подтверждает регион. Автоподтверждение запрещено.</div>`;
  }
  let rgAction;
  if (rgSource === 'explicit_text') {
    rgAction = 'Регион подтверждён (text evidence' + (rgCanonMatch ? ' + canonical match' : '') + ').';
  } else if (rgSource === 'brand_region_whitelist') {
    rgAction = 'Регион подтверждён правилом brand → region whitelist: ' + rgBrandInf.brand + ' → ' + rgBrandInf.inferredRegion + '.';
  } else if (rgSource === 'canonical_alias' && rgStatus === 'green') {
    rgAction = 'Регион подтверждён (canonical match).';
  } else if (rgSource === 'canonical_alias' && rgStatus === 'yellow') {
    rgAction = rgMod.reason || 'Canonical match, но seller text не подтверждает. Требуется ручная проверка.';
  } else {
    rgAction = rgMod.reason || '—';
  }
  const rgSourceLabel = rgSource === 'brand_region_whitelist'
    ? ` <span style="color:#a78bfa;font-size:11px;">(brand whitelist: ${esc(rgBrandInf.brand)} → ${esc(rgBrandInf.inferredRegion)})</span>`
    : (rgCanonMatch && rgCand2 ? ' <span style="color:#4ade80;font-size:11px;">(canonical match)</span>' : '');
  const sec4 = renderDetailSection('Регион', 4, rgStatus, [
    { label: 'Текущий', value: rgCurrent },
    { label: 'Рекомендованный', html: esc(rgCand2 || '—') + rgSourceLabel + rgHints },
    { label: 'Действие', value: rgAction },
    { label: 'Источник', value: rgSource === 'brand_region_whitelist' ? 'brand whitelist' : 'catalogue' },
  ]);

  /* Block 5: Denomination (MSG87 + MSG135: text-evidence gate)
     Rules (see LIST_CHECK_STATUS_RULES_REGISTRY):
       green   = recommended found AND current canonically equals recommended AND text evidence confirmed
       yellow  = recommended found but current differs OR text evidence missing OR region unconfirmed
       red     = explicit mismatch / no valid candidate
       neutral = ONLY if no recommended candidate exists at all */
  const dnMod = mods.denomination || {};
  const dnCand = sel.denomination_candidate_id || '';
  const dnCurrent = den2(b);
  const dnCanonMatch = denominationsCanonicallyEqual(dnCurrent, dnCand);
  /* MSG135: text-evidence gate for denomination */
  const dnTextConfirmed = dnMod.status === 'match';
  const dnUncertain = dnMod.status === 'uncertain';
  /* MSG135 + MSG178: denomination depends on resolved region status, not raw module status.
     Brand inference can resolve region to green even when module says uncertain. */
  const dnRegionBlocked = rgStatus === 'yellow' || rgStatus === 'red';
  let dnStatus;
  if (dnCand && dnCanonMatch && dnTextConfirmed && !dnRegionBlocked) {
    dnStatus = 'green';                      /* canonical + text evidence + region OK */
  } else if (dnCand && dnCanonMatch && (dnUncertain || dnRegionBlocked)) {
    dnStatus = 'yellow';                     /* canonical OK but evidence missing or region blocked */
  } else if (dnCand && dnCanonMatch) {
    dnStatus = 'green';                      /* canonical match, no uncertainty flags */
  } else if (dnCand && !dnCanonMatch) {
    dnStatus = 'yellow';                     /* recommended found, current differs → apply needed */
  } else if (dnMod.status === 'mismatch') {
    dnStatus = 'red';                        /* explicit conflict */
  } else if (!dnCand && dnTextConfirmed) {
    dnStatus = 'green';                      /* match without specific candidate (backend confirmed) */
  } else if (!dnCand) {
    dnStatus = 'gray';                       /* no candidate, no conflict → truly neutral */
  } else {
    dnStatus = 'yellow';                     /* fallback: candidate exists → needs review */
  }
  let dnAction;
  if (dnStatus === 'green') {
    dnAction = dnMod.reason || 'Denomination подтверждён.';
  } else if (dnCanonMatch && dnRegionBlocked) {
    dnAction = 'Denomination совпадает с candidate, но регион не подтверждён seller text. Автоподтверждение denomination заблокировано.';
  } else if (dnCanonMatch && dnUncertain) {
    dnAction = dnMod.reason || 'Denomination совпадает с candidate, но text evidence отсутствует. Требуется ручная проверка.';
  } else if (dnStatus === 'yellow') {
    dnAction = 'Рекомендованный denomination найден, но текущее значение в карточке отличается. Нужно применить замену.';
  } else if (dnStatus === 'red') {
    dnAction = dnMod.reason || 'Конфликт denomination.';
  } else {
    dnAction = dnMod.reason || '—';
  }
  const sec5 = renderDetailSection('Denomination', 5, dnStatus, [
    { label: 'Текущий', value: dnCurrent },
    { label: 'Режим', value: dnMod.mode || 'numeric' },
    { label: 'Рекомендованный', value: dnCand || dnCurrent },
    { label: 'Действие', value: dnAction },
    { label: 'Источник', value: 'catalogue' },
  ]);

  /* Block 1: Итоговый результат (aggregated from child blocks — MSG77) */
  const childStatuses = [mrStatus, ptStatus, rgStatus, dnStatus];
  const hasRed    = childStatuses.includes('red');
  const hasYellow = childStatuses.includes('yellow');
  const aggStatus = hasRed ? 'red' : hasYellow ? 'yellow' : 'green';
  const aggC = clr(aggStatus);
  const aggAction = aggStatus === 'green' ? 'можно сохранить без ручной проверки'
    : aggStatus === 'red' ? 'отклонить или исправить вручную'
    : 'проверить рекомендации и применить';
  const sec1 = renderDetailSection('Итоговый результат', 1, aggStatus, [
    { label: 'Название', value: card.productName },
    { label: 'Итоговый статус', value: aggC.label },
    { label: 'Главная причина', value: card.aiReason || '—' },
    { label: 'Необходимо', value: aggAction },
  ]);

  /* Block 6: State badges + errors + manual confirm */
  const blockIssues = getCardBlockingIssues(card);
  let stateHtml = '<div style="margin-top:8px;">';
  if (card.applied) stateHtml += '<span style="background:rgba(99,102,241,.25);color:#a5b4fc;padding:3px 8px;border-radius:6px;font-size:12px;margin-right:4px;">Applied</span>';
  if (card.saved) stateHtml += '<span style="background:rgba(34,197,94,.2);color:#4ade80;padding:3px 8px;border-radius:6px;font-size:12px;margin-right:4px;">Saved</span>';
  if (card.exportedToSheet) stateHtml += '<span style="background:rgba(56,189,248,.2);color:#38bdf8;padding:3px 8px;border-radius:6px;font-size:12px;margin-right:4px;">В таблице</span>';
  if (card.modStatusApplied) stateHtml += '<span style="background:rgba(245,158,11,.2);color:#fcd34d;padding:3px 8px;border-radius:6px;font-size:12px;">MS:' + esc(card.modStatusNew || '?') + '</span>';
  if (card.manuallyConfirmed) stateHtml += '<div style="color:#4ade80;margin-top:6px;font-size:12px;">✓ Ручные правки подтверждены</div>';
  if (card.scanError) stateHtml += '<div style="color:#f87171;margin-top:4px;font-size:12px;">⚠ scan: ' + esc(card.scanError) + '</div>';
  if (card.applyError) stateHtml += '<div style="color:#fb923c;margin-top:4px;font-size:12px;">⚠ apply: ' + esc(card.applyError) + '</div>';
  if (card.saveError) stateHtml += '<div style="color:#f87171;margin-top:4px;font-size:12px;">⚠ save: ' + esc(card.saveError) + '</div>';
  if (blockIssues.length > 0 && !card.manuallyConfirmed) {
    stateHtml += `<div style="margin-top:8px;">
      <button type="button" data-confirm-detail="${esc(card.productId)}" style="border:none;border-radius:8px;
        padding:8px 16px;background:#d97706;color:#fff;font-size:13px;font-weight:700;cursor:pointer;">
        Подтвердить ручные правки</button>
      <span style="font-size:11px;color:#9ca3af;margin-left:8px;">Исправьте поля в popup, затем нажмите</span>
    </div>`;
  }
  stateHtml += '</div>';

  /* Color override */
  const ovHtml = ['green','yellow','red'].map(col => {
    const active = card.operatorOverride === col;
    return `<button type="button" data-ov-detail="${col}" style="border:none;border-radius:6px;padding:4px 10px;
      cursor:pointer;font-size:14px;opacity:${active ? '1' : '.35'};
      background:${active ? 'rgba(255,255,255,.15)' : 'transparent'};">${clr(col).badge}</button>`;
  }).join('');

  detail.innerHTML = `
    <div style="border:1px solid rgba(255,255,255,.1);border-radius:20px;background:#021132;overflow:hidden;">
      <div style="padding:16px 20px;border-bottom:1px solid rgba(255,255,255,.1);">
        <div style="display:flex;align-items:center;justify-content:space-between;gap:12px;">
          <div>
            <div style="font-size:11px;color:rgba(255,255,255,.4);text-transform:uppercase;letter-spacing:.15em;">Detail View</div>
            <div style="font-size:20px;font-weight:800;margin-top:2px;">${esc(card.productName)}</div>
            <div style="font-size:12px;color:rgba(255,255,255,.5);margin-top:3px;">ID ${esc(card.productId)} · ${esc(card.aiReason || '')}</div>
          </div>
          <div style="display:flex;align-items:center;gap:6px;">
            <span style="display:inline-flex;align-items:center;gap:5px;border:1px solid ${c.border};
              background:${c.bg};color:${c.txt};padding:4px 12px;border-radius:14px;font-size:13px;font-weight:700;">
              ${c.badge} ${c.label}</span>
            <div style="display:flex;gap:2px;">${ovHtml}</div>
          </div>
        </div>
      </div>
      <div style="padding:14px 18px;">
        ${ticketCtx}
        <div style="display:grid;grid-template-columns:1fr 1fr;gap:8px;">
          <div style="grid-column:1/-1;">${sec1}</div>
          ${sec2}
          ${sec3}
          ${sec4}
          ${sec5}
        </div>
        ${stateHtml}
        ${renderAiTraceBlock(card, session)}
      </div>
    </div>`;

  /* Bind trace toggle */
  const traceToggle = $('wip01-trace-toggle');
  if (traceToggle) traceToggle.onclick = () => {
    _s.uiTraceOpen = !_s.uiTraceOpen;
    renderDetailPanel(session);
  };

  /* Bind detail-level buttons */
  $$('[data-confirm-detail]').forEach(btn => {
    btn.onclick = async () => {
      btn.disabled = true; btn.textContent = 'Проверяю popup...';
      try { await confirmManualFixes(card); } catch (e) {
        card.saveError = (card.saveError ? card.saveError + '; ' : '') + (e.message || String(e));
      }
      renderCardList(session); renderDetailPanel(session);
    };
  });
  $$('[data-ov-detail]').forEach(btn => {
    btn.onclick = () => {
      const col = btn.dataset.ovDetail;
      card.operatorOverride = (card.operatorOverride === col) ? null : col;
      rebuildSummary(session); renderCardList(session); renderDetailPanel(session);
      renderActionBar(session);
    };
  });
}

/* ── TOP ACTION BAR ─────────────────────────────────────────── */
function renderActionBar(session) {
  const all   = session.cards;
  const sel   = _s.uiSelectedId ? all.find(c => c.productId === _s.uiSelectedId) : null;
  const yellows     = all.filter(c => c.finalColor === 'yellow' || c.finalColor === 'red');
  const applyable   = all.filter(c => (c.finalColor === 'yellow' || c.finalColor === 'red') && !c.applied);
  const saveReady   = all.filter(c => isCardSaveReady(c));
  const yellowRedForSheet = all.filter(c => (c.finalColor === 'yellow' || c.finalColor === 'red') && !c.exportedToSheet);

  const btn = (id, label, enabled, bg, col) =>
    `<button type="button" id="${id}" ${enabled ? '' : 'disabled'} style="border:none;border-radius:10px;
      padding:8px 14px;font-size:12px;font-weight:700;cursor:${enabled ? 'pointer' : 'default'};
      background:${enabled ? bg : 'rgba(255,255,255,.05)'};color:${enabled ? col : '#4b5563'};
      transition:background .15s;">${label}</button>`;

  $('wip01-actions').innerHTML = [
    btn('wip01-apply1', 'Применить рекомендации', !!sel && !sel.applied && sel.recommendations, 'rgba(56,189,248,.15)', '#38bdf8'),
    btn('wip01-applyAll', 'Применить для всех', applyable.length > 0, 'rgba(56,189,248,.08)', '#7dd3fc'),
    btn('wip01-save', 'Сохранить изменения', saveReady.length > 0, 'rgba(34,197,94,.15)', '#4ade80'),
    btn('wip01-sheet1', 'В таблицу', !!sel, 'rgba(99,102,241,.15)', '#a5b4fc'),
    btn('wip01-sheetAll', 'В таблицу всё', yellowRedForSheet.length > 0, 'rgba(99,102,241,.08)', '#c4b5fd'),
    btn('wip01-ms', 'Mod Status', all.length > 0, 'rgba(168,85,247,.15)', '#c084fc'),
  ].join('');

  /* Bind */
  const b = id => $(id);

  /* Применить рекомендации — одна выбранная карточка (MSG89: always use fresh sel) */
  const a1 = b('wip01-apply1');
  if (a1 && !a1.disabled) a1.onclick = async () => {
    const freshSel = _s.uiSelectedId ? all.find(c => c.productId === _s.uiSelectedId) : null;
    if (!freshSel) return;
    if (freshSel.applied) { renderActionBar(session); return; }
    a1.disabled = true; a1.textContent = 'Применяю...';
    _s.isFirstCard = true;
    const snapBefore = freshSel.afterSnapshot || freshSel.beforeSnapshot;
    try { await applyRecommendations(freshSel); } catch (e) {
      freshSel.applyError = (freshSel.applyError ? freshSel.applyError + '; ' : '') + (e.message || String(e));
    }
    /* Verify that something actually changed (MSG123: consistent applied state) */
    const snapAfter = freshSel.afterSnapshot;
    if (snapBefore && snapAfter) {
      const ptChanged2 = (snapBefore.productTypeId !== snapAfter.productTypeId);
      const rgChanged2 = JSON.stringify((snapBefore.regions||[]).map(r=>r.id).sort()) !== JSON.stringify((snapAfter.regions||[]).map(r=>r.id).sort());
      const dnChanged2 = (snapBefore.denominationOptionId !== snapAfter.denominationOptionId);
      const dnAlready2 = ((freshSel.recommendations || {}).denomination || {}).alreadyMatched === true;
      if (!ptChanged2 && !rgChanged2 && !dnChanged2 && !dnAlready2 && !freshSel.applyError) {
        freshSel.applyError = (freshSel.applyError ? freshSel.applyError + '; ' : '') +
          'APPLY_NO_CHANGE: Popup открыт, но ни одно поле не изменилось. Проверьте рекомендации вручную.';
      }
    }
    /* MSG123 ACTION 2: APPLY_NO_CHANGE + applied=true is inconsistent */
    if (freshSel.applyError && freshSel.applyError.includes('APPLY_NO_CHANGE')) {
      freshSel.applied = false;
    }
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };

  /* Применить для всех */
  const aa = b('wip01-applyAll');
  if (aa && !aa.disabled) aa.onclick = async () => {
    aa.disabled = true; aa.textContent = 'Применяю...';
    _s.isFirstCard = true;
    for (const card of applyable) {
      if (_s.aborted) break;
      try { await applyRecommendations(card); } catch (e) {
        card.applyError = (card.applyError ? card.applyError + '; ' : '') + (e.message || String(e));
      }
      renderCardList(session);
      await wait(150);
    }
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };

  /* Сохранить изменения */
  const sv = b('wip01-save');
  if (sv && !sv.disabled) sv.onclick = async () => {
    sv.disabled = true; sv.textContent = 'Сохраняю...';
    _s.isFirstCard = true;
    for (const card of saveReady) {
      if (_s.aborted) break;
      if (!isCardSaveReady(card)) {
        card.saveError = 'SAVE_BLOCKED: ' + (getCardBlockingIssues(card).join(', ') || 'not ready');
        continue;
      }
      try { await saveCard(card); } catch (e) { card.saveError = e.message || String(e); }
      renderCardList(session);
      await wait(300);
    }
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };

  /* В таблицу — одна карточка */
  const s1 = b('wip01-sheet1');
  if (s1 && !s1.disabled) s1.onclick = async () => {
    if (!sel) return;
    await exportToSheet(session, [sel]);
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };

  /* В таблицу всё — yellow + red */
  const sa = b('wip01-sheetAll');
  if (sa && !sa.disabled) sa.onclick = async () => {
    await exportToSheet(session, yellowRedForSheet);
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };

  /* Mod Status */
  const ms = b('wip01-ms');
  if (ms && !ms.disabled) ms.onclick = async () => {
    /* Yellow → send to sheet (no mod status change), green → published, red → rejected */
    const yellowCards = all.filter(c => c.finalColor === 'yellow' && !c.exportedToSheet);
    if (yellowCards.length > 0) {
      await exportToSheet(session, yellowCards);
    }
    for (const card of all) {
      if (_s.aborted) break;
      const rule = getModStatusRule(card.finalColor);
      if (!rule || rule.modStatus === 'no_change') continue;
      if (card.modStatusApplied) continue;
      try { await applyModStatus(card); } catch (e) {
        card.saveError = (card.saveError ? card.saveError + '; ' : '') + 'modStatus: ' + (e.message || String(e));
      }
      renderCardList(session);
      await wait(200);
    }
    renderCardList(session); renderDetailPanel(session); renderActionBar(session);
  };
}

/* ── FULL DASHBOARD RENDER ──────────────────────────────────── */
function renderDashboard(session) {
  renderCardList(session);
  renderDetailPanel(session);
  renderActionBar(session);
}

/* ── MAIN ────────────────────────────────────────────────────── */
async function main() {
  /* MSG158: load alias snapshot before anything else */
  await loadAliasSnapshot();

  const items   = scanUnderReviewRows();
  const session = makeBatchSession(items.length);
  _s.session    = session;
  _s.uiSelectedId = null;

  renderOverlay();

  /* MSG129: doScan now accepts an array of selected items, not a count */
  async function doScan(selectedItems) {
    _s.aborted = false;
    _s.uiPhase = 'scanning';
    const toScan = selectedItems && selectedItems.length > 0 ? selectedItems : items;
    session.summary.totalFound = toScan.length;
    renderProgress(0, toScan.length, null);

    await runScan(toScan, session, (done, total, card) => {
      renderProgress(done, total, card);
    });

    /* ── PHASE TRANSITION: scanning → results/empty ── */
    if (session.cards.length > 0) {
      _s.uiSelectedId = session.cards[0].productId;
      _s.uiPhase = 'results';
    } else {
      _s.uiPhase = 'empty';
    }
    renderDashboard(session);
  }

  renderStartup(items, doScan);
}

main().catch(e => {
  console.error('[BSV WIP01]', e);
  const d = $('wip01-detail');
  if (d) d.innerHTML = '<div style="color:#f87171;padding:16px;">Критическая ошибка: ' + esc(e.message || String(e)) + '</div>';
});

})();
